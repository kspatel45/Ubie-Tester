// Continuous Read Loop (Fixed Stream Locking) async function readLoop() { keepReading = true; while (port.readable && keepReading) { try { // Pipe the serial input stream directly into a text decoder stream const textDecoder = new TextDecoderStream(); const readableStreamClosed = port.readable.pipeTo(textDecoder.writable); reader = textDecoder.readable.getReader(); while (keepReading) { const { value, done } = await reader.read(); if (done) { break; } if (value) { // Log raw data chunks as they arrive from the COM port logTraffic('RECEIVED', value); } } } catch (error) { console.error("Read error:", error); // Append error to traffic log for better visibility logTraffic('ERROR', error.message); } finally { if (reader) { reader.releaseLock(); } } // Brief pause before trying to reconnect the reader loop if interrupted await new Promise(resolve => setTimeout(resolve, 100)); } }
